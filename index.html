// ========================================================================= // CHỨC NĂNG ĐỒNG BỘ CÀO NGẦM GULAND - CHẠY CHUNG CHO CẢ PC VÀ ĐIỆN THOẠI (LEMUR) // ========================================================================= layer.on('click', function(e) { var soTo = properties.so_to; var soThua = properties.so_thua; var lat = layer.getBounds().getCenter().lat; var lng = layer.getBounds().getCenter().lng; var gulandUrl = `https://guland.vn/ban-do-quy-hoach?lat=${lat}&lng=${lng}&zoom=19`; // 1. Tạo thẻ iframe ngầm var iframe = document.createElement('iframe'); iframe.src = gulandUrl; iframe.name = JSON.stringify({ so_to: soTo, so_thua: soThua }); iframe.style.border = 'none'; // 2. TỰ ĐỘNG NHẬN DIỆN THIẾT BỊ ĐỂ ÁP DỤNG CẤU HÌNH PHÙ HỢP var isMobile = /Android|iPhone|iPad|iPod/i.test(navigator.userAgent); if (isMobile) { // Cấu hình dành riêng cho Điện thoại (Lemur Browser) để tránh bị đóng băng ngầm iframe.style.width = '1px'; iframe.style.height = '1px'; iframe.style.position = 'absolute'; iframe.style.left = '-9999px'; console.log("Đang chạy luồng ngầm trên Điện thoại (Lemur)..."); } else { // Cấu hình dành riêng cho Máy tính (PC) - Ẩn hoàn toàn cực sạch sẽ iframe.style.display = 'none'; console.log("Đang chạy luồng ngầm trên Máy tính (PC)..."); } // 3. Đút khung vào trang web để kích hoạt cào dữ liệu document.body.appendChild(iframe); // 4. Thời gian chờ tự hủy khung (Ưu tiên để 12 giây cho di động, PC xong trước vẫn tự hủy được) setTimeout(function() { if (iframe) { iframe.remove(); console.log("Đã hủy khung ngầm của thửa: " + soThua); } }, 12000); });
